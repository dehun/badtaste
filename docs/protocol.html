3 rooms

3.1 join to tagged room queue (game in your city)

-> JoinTaggedRoomQueue(tag = "kiev") <- OnJoinedToTaggedRoomQueue(tag = "kiev") <- OnJoinedToRoom(users = [guid1, guid2, guid3], state = ("pending" | "active"))

3.2 join to a random room

-> JoinMainRoomQueue() <- OnJoinedToMainRoomQueue() <- OnJoinedToRoom(users = [guid1, guid2, guid3], state = ("pending" | "active"))

3.3 on room dead(all users leaved room)

<- OnRoomDeath()

3.4 on somebody joined to room when you are in this room

<- OnRoomUserListChanged(users = [guid1, guid2, guid3])

3.5 on room came to active state(game begin)

<- OnRoomStateChaned(state = "active")

4 chat

4.1 send chat message to room where you are

-> SendChatMessageToRoom(message = "some message text body")

4.2 getting chat message from room where you are

<- OnGotChatMessageFromRoom(senderGuid = "someguid", message = "some message text body")

4.3 send vip chat message to room where you are

-> SendVipChatMessageToRoom(message = "some message text body")

4.4 getting vipchat message from room where you are

<- OnGotVipChatMessageFromRoom(senderGuid = "someguid", message = "some message text body")

6 user info

6.1 get user info

-> GetUserInfo(targetUserGuid = "someGuid") <- OnGotUserInfo(infoOwnerGuid = "someGuid", userId = "someid", firstName = "lol", lastName = "what", profileUrl = "url", isMan = "false", smallAvatarUrl = "anotherurl", mediumAvatarUrl = "yepanotherurl", bigAvatarUrl = "urrrl", isNameHidden="false", isSocialInfoHidden = "false", isCityHidden="false")

6.2 get friend info

-> GetFriendInfo(targetUserGuid = "someGuid") <- OnGotUserInfo(infoOwnerGuid = "someGuid", userId = "someid", firstName = "lol", lastName = "what", profileUrl = "url", isMan = "false", smallAvatarUrl = "anotherurl", mediumAvatarUrl = "yepanotherurl", bigAvatarUrl = "urrrl")

6.3 upload avatar (by http)

-> UploadNewUserAvatar(userGuid = "someguid", imageFormatName = "png", imageDataBase64 = "imagedataencodedintobase64string") <- UploadNewUserAvatarResult(result = "ok")

6.4 update user info from game

9 rates

9.1 rate user success

-> RateUser(targetUserGuid = "someguid", rate = 5) <- OnUserRatedSuccessfully(ratedUserGuid = "someguid")

9.2 rate user fail

-> RateUser(targetUserGuid = "someguid", rate = 5) <- OnUserRateFailed(ratedUserGuid = "someguid", reason = "alreadyrated")

9.3 get user rate

-> GetUserRate(targetUserGuid = "someguid") <- OnGotUserRate(userGuid = "someguid", averageRate = 4, lastRaters = [Rater(guid = "some", rate = 2), Rater(guid = "other", rate = 10)])

9.4 remove user rate succcess

-> DeleteRatePoint(raterGuid = "someguid") <- OnRatePointDeleted(raterGuid = "someguid")

9.5 remove user rate failure

-> DeleteRatePoint(raterGuid = "someguid") <- OnRatePointDeleteFailed(raterGuid = "someguid")

9.6 are user rated

-> AreUserRated(targetUserGuid = "someguid") <- OnGotAreUserRated(targetUserGuid = "someguid", areRated = "true")

10 gifts

10.2 when gift presented to somebody in room

-> PresentGift(targetUserGuid, giftGuid) <- OnGotGift(giftSenderGuid, giftGuid) <- OnGiftReceivedInGame(giftSenderGuid, giftReceiverGuid, giftGuid)

10.3 get gift list

-> GetMyGifts() <- OnGotMyGifts(gifts = [Gift(senderGuid = "123", giftGuid = "234")])

10.4 get gift list of me or friend

-> GetUserGifts(targetUserGuid = "someGuid") <- OnGotUserGifts(ownerGuid = "someGuid", gifts = [Gift(senderGuid = "123", giftGuid = "234")])
